func echo(numbers []int) <-chan int {
	out := make(chan int)
	go func() {
		for index := range numbers {
			out <- numbers[index]
		}
		close(out)
	}()
	return out
}

func sq(in <-chan int) <-chan int {
	out := make(chan int)
	go func() {
		for n := range in {
			out <- n * n
		}
		close(out)
	}()
	return out
}

func addOne(in <-chan int) <-chan int {
	out := make(chan int)
	go func() {
		for n := range in {
			out <- n + 1
		}
		close(out)
	}()
	return out
}

func main() {
	for out := range addOne(sq(echo([]int{1, 2, 3}))) {
		fmt.Println(out)
	}
}
